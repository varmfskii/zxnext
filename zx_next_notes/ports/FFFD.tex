\port{FFFD}{AY Control and AY register Select} (3.01.09)\\
Read
\begin{itemize}
\item[] bits 7-0 = data in selected register of active AY chip
\end{itemize}
Select Chip
\begin{itemize}
\item[] bit 7 = 1
\item[] bit 6 = Enable left
\item[] bit 5 = Enable Right
\item[] bits 4-2 = Reserved, must be 1
\item[] bits 1-0 = AY chip select
  \begin{itemize}
  \item[] 00 = Unused
  \item[] 01 = AY 2
  \item[] 10 = AY 1
  \item[] 11 = AY 0
  \end{itemize}
\end{itemize}
Select Register
\begin{itemize}
\item[] bit 7 = 0
\item[] bits 6-5 = Reserved, must be 0
\item[] bits 4-0 = Register Number
\end{itemize}
