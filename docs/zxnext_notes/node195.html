File calls

f_open $9A (9A) open file        
f_close $9B (9B) close file        
f_sync $9C (9C) sync file changes to disk        
f_read $9D (9D) read file        
f_write $9E (9E) write file        
f_seek $9F (9F) set file position        
f_fgetpos $A0 (A0) get file position        
f_fstat $A1 (A1) get open file information        
f_ftruncate $A2 (A2) truncate/extend open file        
f_opendir $A3 (A3) open directory for reading        
f_readdir $A4 (A4) read directory entry        
f_telldir $A5 (A5) get directory position        
f_seekdir $A6 (A6) set directory position        
f_rewinddir $A7 (A7) rewind to start of directory        
f_getcwd $A8 (A8) get current working directory        
f_chdir $A9 (A9) change directory        
f_mkdir $AA (AA) make directory        
f_rmdir $AB (AB) remove directory        
f_stat $AC (AC) get unopen file information        
f_unlink $AD (AD) delete file        
f_truncate $AE (AE) truncate/extend unopen file        
f_chmod $AF (AF) change file attributes        
f_rename $B0 (B0) rename/move file        
f_getfree $B1 (B1) get free space
